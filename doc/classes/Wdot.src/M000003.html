  def Wdot.start_node_parse(line)
    line.match Start_node_pat
    full_node_name = $1
    node_name = $2
    node_name = $3 if $3
    parse_string = "\#{full_node_name} [label=\"\#{node_name}\", shape=\"\#{@@start_node_shape}\",\n  fillcolor=\"\#{@@start_node_fillcolor}\"]\n"
  end
